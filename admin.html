bookBtn.addEventListener('click', async () => { const room = document.getElementById('room').value; const date = document.getElementById('date').value; const start_time = document.getElementById('start_time').value; const duration_min = parseInt(document.getElementById('duration_min').value, 10); if (!room || !date || !start_time || !duration_min) { msg.style.color = 'red'; msg.textContent = 'Bitte alle Felder ausfüllen.'; return; } try { const res = await fetch('/add-slot', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ room, date, start_time: date + 'T' + start_time + ':00', // ISO-Format duration_min }) }); const j = await res.json(); if (j.success) { msg.style.color = 'green'; msg.textContent = 'Slot erfolgreich hinzugefügt (ID ' + j.id + ')'; form.reset(); } else { msg.style.color = 'red'; msg.textContent = j.error || 'Fehler beim Hinzufügen'; } } catch (e) { console.error(e); msg.style.color = 'red'; msg.textContent = 'Serverfehler'; } });
